SQL — SELECT ו-WHERE

⏱ זמן משוער: 40 דק'

מקור: sql_intro.pdfמקור: SQL-cheat-sheet.pdf

מבנה שאילתת SELECT

שאילתת שליפה בסיסית:

SELECT column1, column2   -- אילו עמודות
FROM   table_name         -- מאיזו טבלה
WHERE  condition          -- אילו שורות
ORDER BY column [ASC|DESC] -- מיון
LIMIT  n;                 -- כמה שורות

SELECT * מחזיר את כל העמודות; רשימת עמודות מופרדת בפסיקים בוחרת ספציפיות. DISTINCT מסיר שורות כפולות מהתוצאה. אפשר לתת כינוי (alias) לעמודה: SELECT first_name fname FROM actor (או עם AS). הערות ב-SQL: -- הערה עד סוף השורה.

דוגמה — הטבלה הזו תלווה אותנו בכל הפרק:

-- employee:
-- emp_id | first_name | salary | city     | dept_id
--   1    | Dana       | 12000  | Tel Aviv | 10
--   2    | Avi        | 6500   | Haifa    | 10
--   3    | Noa        | 9000   | Tel Aviv | 20
--   4    | Yossi      | 4000   | Haifa    | NULL

SELECT * FROM employee;                 -- כל 4 השורות, כל העמודות
SELECT first_name, salary FROM employee; -- 4 שורות, 2 עמודות

SELECT DISTINCT city FROM employee;
-- תוצאה (בלי כפילויות):
-- Tel Aviv
-- Haifa

SELECT first_name AS name, salary * 12 AS yearly_salary
FROM employee;                          -- כינויים לעמודות מחושבות
-- Dana | 144000
-- Avi  | 78000  ...

סדר כתיבה: SELECT → FROM → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT. סדר ביצוע לוגי מתחיל דווקא מ-FROM ואז WHERE, ורק בסוף SELECT ו-ORDER BY.

WHERE ואופרטורים

  • השוואה: = < > <= >= <> (או !=).
  • לוגיים: AND, OR, NOT (שימו לב לסוגריים בשילוב).
  • טווח: BETWEEN a AND b. רשימה: IN (v1, v2, ...).
  • NULL: IS NULL / IS NOT NULL — לעולם לא = NULL.

מחרוזות עוטפים תמיד בגרש בודד: WHERE name = 'Thailand'; מספרים נכתבים בלי גרש: WHERE population > 5000000.

על טבלת ה-employee מלמעלה:

-- תבנית: SELECT columns FROM table WHERE condition;

SELECT first_name, salary FROM employee
WHERE salary > 8000;
-- Dana (12000), Noa (9000)

SELECT first_name FROM employee
WHERE salary BETWEEN 5000 AND 10000;    -- כולל את הקצוות
-- Avi (6500), Noa (9000)

SELECT first_name FROM employee
WHERE city = 'Tel Aviv' AND salary > 10000;
-- Dana בלבד (נועה בת"א אבל מתחת ל-10000)

SELECT first_name FROM employee
WHERE dept_id IN (10, 30);              -- שקול ל: dept_id=10 OR dept_id=30
-- Dana, Avi

SELECT first_name FROM employee
WHERE dept_id IS NULL;                  -- = NULL לא היה תופס כלום!
-- Yossi

LIKE ותבניות

LIKE מחפש תבניות טקסט. תווים כלליים:

  • % — רצף תווים כלשהו (כולל ריק). 'A%' = מתחיל ב-A; '%a' = מסתיים ב-a; '%son%' = מכיל son.
  • _ — בדיוק תו אחד. '_o_' = מילה בת 3 עם o באמצע.

על השמות Dana, Avi, Noa, Yossi:

-- תבנית: WHERE column LIKE 'pattern';   (% = כל רצף, _ = תו אחד)

SELECT first_name FROM employee WHERE first_name LIKE 'D%';
-- Dana                          (מתחיל ב-D)

SELECT first_name FROM employee WHERE first_name LIKE '%a';
-- Dana, Noa                     (מסתיים ב-a)

SELECT first_name FROM employee WHERE first_name LIKE '%ss%';
-- Yossi                         (מכיל ss)

SELECT first_name FROM employee WHERE first_name LIKE '_o_';
-- Noa                           (בדיוק 3 תווים, o באמצע)

SELECT first_name FROM employee WHERE first_name LIKE '____';
-- Dana                          (בדיוק 4 תווים)

ORDER BY ו-LIMIT

ORDER BY salary DESC, name ASC ממיין קודם לפי שכר יורד ואז לפי שם עולה. LIMIT 5 מחזיר 5 שורות; LIMIT 5 OFFSET 10 (או LIMIT 10, 5) מדלג על 10 ומחזיר את הבאים.

-- תבנית: SELECT ... ORDER BY col [ASC|DESC] LIMIT n [OFFSET m];

SELECT first_name, salary FROM employee
ORDER BY salary DESC;
-- Dana  | 12000
-- Noa   | 9000
-- Avi   | 6500
-- Yossi | 4000

SELECT first_name, salary FROM employee
ORDER BY salary DESC
LIMIT 2;                      -- שני המרוויחים הגדולים
-- Dana | 12000
-- Noa  | 9000

SELECT first_name, salary FROM employee
ORDER BY salary DESC
LIMIT 2 OFFSET 1;             -- מדלג על הראשון, מחזיר את ה-2 הבאים
-- Noa | 9000
-- Avi | 6500

🧪 מעבדת SQL — תרגול חי

כתבו SQL אמיתי והריצו אותו כאן, ישירות בדפדפן. ▶ הרצה מציגה את התוצאה שלכם; בדיקה מול פתרון משווה אותה לתוצאת הפתרון הרשמי.

המנוע (SQLite ב-WebAssembly) רץ כולו בדפדפן ותואם ל-MySQL בכל מה שנלמד בקורס — AUTO_INCREMENT, DESCRIBE ו-TRUNCATE מתורגמים אוטומטית. כל הרצה מתחילה ממסד נקי, אז אי-אפשר לקלקל כלום.

🧪 משימה: שלפו שם ושכר של העובדים שמרוויחים יותר מ-8000. (הטבלה זהה לדוגמאות בפרק.)

SELECT first_name, salary FROM employee
WHERE salary > 8000;

🧪 משימה: שלפו את שמות העובדים מתל אביב ('Tel Aviv') ששכרם בין 8000 ל-13000 — עם BETWEEN.

SELECT first_name FROM employee
WHERE city = 'Tel Aviv' AND salary BETWEEN 8000 AND 13000;

🧪 משימה: שלפו את שני העובדים עם השכר הגבוה ביותר (שם + שכר), ממוינים מהגבוה לנמוך.

SELECT first_name, salary FROM employee
ORDER BY salary DESC
LIMIT 2;
