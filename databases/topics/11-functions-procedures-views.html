Views

View הוא שאילתה שמורה שמתנהגת כטבלה וירטואלית — אינו מאחסן נתונים אלא מריץ את השאילתה בכל פנייה. שימושי להפשטה, אבטחה (חשיפת עמודות מסוימות) ושימוש חוזר:

CREATE VIEW high_earners AS
SELECT first_name, salary FROM employee
WHERE salary > 10000;

SELECT * FROM high_earners;   -- כמו טבלה
SELECT * FROM high_earners h        -- View משתתף גם ב-JOIN וב-WHERE
JOIN department d ON h.dept_id = d.dept_id
WHERE h.salary < 20000;

CREATE VIEW v2 (name, total) AS SELECT …;  -- קביעת שמות עמודות ל-View
DROP VIEW high_earners;                    -- מחיקת ה-View (לא הנתונים!)

שתי תוספות שמופיעות בדף-העזר של המרצה: CREATE TEMPORARY VIEW — View זמני שנעלם מעצמו כשסוגרים את החיבור (session); ו-WITH CHECK OPTION — כשמעדכנים/מוסיפים שורות דרך ה-View, האופציה חוסמת שינוי שסותר את תנאי ה-WHERE של ה-View (למשל הוספת עובד עם שכר 5,000 דרך View שמציג רק שכר מעל 10,000).

Stored Functions

פונקציה מאוחסנת מקבלת פרמטרים ומחזירה ערך יחיד (RETURNS), ושמישה בתוך שאילתות:

CREATE FUNCTION annual_salary(m DECIMAL(10,2))
RETURNS DECIMAL(10,2)
DETERMINISTIC
RETURN m * 12;

SELECT first_name, annual_salary(salary) FROM employee;

מה זה DETERMINISTIC? הצהרה («מאפיין הגדרה») שאומרת ל-MySQL: לאותו קלט הפונקציה מחזירה תמיד את אותו פלט — חישוב טהור, בלי אקראיות, בלי תלות בזמן ובלי קריאה מטבלאות. annual_salary באמת כזו (m*12 תלוי רק ב-m). ההצהרה מאפשרת ל-MySQL לייעל (לשמור תוצאה ולא לחשב שוב) ונדרשת בהגדרות מסוימות של השרת — בלעדיה יצירת הפונקציה עלולה להיכשל.

  • NOT DETERMINISTIC (ברירת המחדל) — פלט שיכול להשתנות גם לאותו קלט, למשל שימוש ב-NOW() או RAND().
  • READS SQL DATA — הפונקציה קוראת נתונים מטבלאות (כמו actor_count_fn שבהמשך): התוצאה תלויה בתוכן הטבלה, ולכן היא אינה דטרמיניסטית — זה המאפיין המתאים לה.

הפונקציה המלאה מהשיעור — גוף עם BEGIN…END, משתנה מקומי והשמה משאילתה. סופרת כמה סרטים יש לשחקן:

CREATE FUNCTION actor_count_fn(in_first_name VARCHAR(45),
                               in_last_name  VARCHAR(45))
RETURNS INTEGER
BEGIN
  DECLARE a_count INTEGER;        -- הגדרת משתנה מקומי
  SELECT COUNT(*) INTO a_count    -- השמה מתוך שאילתה:
  FROM actor a                    -- השאילתה חייבת להחזיר שורה אחת
  JOIN film_actor fa ON a.actor_id = fa.actor_id
  WHERE a.first_name = in_first_name
    AND a.last_name  = in_last_name;
  RETURN a_count;                 -- פונקציה חייבת RETURN
END;

-- דרך נוספת להשמה (בלי שאילתה): SET a_count = 5;
-- וגם משתנה עם ערך התחלתי: DECLARE result INTEGER DEFAULT 0;

שלוש הדרכים מהמצגת להשתמש בפונקציה:

SELECT actor_count_fn('NICK','WAHLBERG');   -- ערך בודד

SELECT first_name, last_name,                -- עמודה מחושבת
       actor_count_fn(first_name, last_name) AS number_of_film
FROM actor;

SELECT first_name, last_name                 -- בתוך WHERE
FROM actor
WHERE actor_count_fn(first_name, last_name) BETWEEN 20 AND 25;

Stored Procedures

פרוצדורה היא בלוק פקודות שמורות שמופעל עם CALL. יכולה לקבל פרמטרים IN (קלט), OUT (פלט) ו-INOUT, ואינה חייבת להחזיר ערך:

CREATE PROCEDURE emp_count(IN dept INT, OUT cnt INT)
BEGIN
  SELECT COUNT(*) INTO cnt FROM employee WHERE dept_id = dept;
END;

CALL emp_count(10, @c);  SELECT @c;

מה זה @c? משתנה עם @ הוא משתנה session של MySQL — משתנה שחי בחיבור (connection) שלכם, לא בתוך טבלה ולא בתוך הפרוצדורה. אין צורך להכריז עליו: הוא נוצר ברגע ההשמה הראשונה (לפני כן ערכו NULL), והוא נשאר קיים עד שסוגרים את החיבור. כשקוראים לפרוצדורה עם פרמטר OUT, חייבים לתת לה «תיבה» לכתוב אליה את התוצאה — ומשתנה ה-session הוא בדיוק התיבה הזו:

SET @c = 5;                    -- השמה ישירה (יוצרת את המשתנה)
SELECT @c;                     -- 5
SELECT @c + 10;                -- שמיש בכל ביטוי: 15

CALL emp_count(10, @c);        -- הפרוצדורה כותבת את התוצאה לתוך @c
SELECT @c;                     -- עכשיו מכיל את מה שהוחזר דרך ה-OUT

SELECT * FROM employee
WHERE salary > @c;             -- ואפשר להמשיך להשתמש בו בשאילתות

שני סוגי משתנים — לא להתבלבל: DECLARE x INT; הוא משתנה מקומי — קיים רק בתוך גוף ה-BEGIN…END של פרוצדורה/פונקציה, וחייב הכרזה עם טיפוס. @x הוא משתנה session — קיים בכל מקום בחיבור, בלי הכרזה ובלי טיפוס קבוע. בקריאה ל-OUT משתמשים דווקא ב-@x כי הוא זה שממשיך לחיות אחרי שהפרוצדורה מסתיימת.

שלושת סוגי הפרמטרים:

  • IN (ברירת המחדל) — קלט בלבד: הקורא מוסר ערך; שינוי שלו בתוך הפרוצדורה לא משפיע על הקורא.
  • OUT — פלט בלבד: הערך הנכנס מתעלמים ממנו (בתוך הפרוצדורה הוא מתחיל כ-NULL); מה שהפרוצדורה כותבת אליו נשאר אצל הקורא אחרי ה-CALL.
  • INOUT — גם וגם: הפרוצדורה מקבלת את הערך של הקורא, יכולה לשנות אותו, והקורא רואה את הערך המעודכן.
CREATE PROCEDURE double_it(INOUT n INT)
BEGIN
  SET n = n * 2;      -- קורא את הערך שנכנס וגם משנה אותו
END;

SET @x = 7;
CALL double_it(@x);   -- נכנס 7, הפרוצדורה מכפילה
SELECT @x;            -- 14 — הקורא רואה את השינוי

הפרוצדורה מהשיעור שמציגה תוצאה ישירות (בלי OUT) — SELECT רגיל בגוף מוצג למי שקרא לה:

CREATE PROCEDURE category_summary_proc(IN in_category_name VARCHAR(20))
BEGIN
  SELECT c.name AS category_name,
         COUNT(*) AS num_of_movies,
         SUM(f.length) AS total_length
  FROM film f
  JOIN film_category fc ON f.film_id = fc.film_id
  JOIN category c       ON fc.category_id = c.category_id
  WHERE c.name = in_category_name
  GROUP BY c.name;
END;

CALL category_summary_proc('action');
-- מציג: 'Action' | 64 | 7143

פונקציה מול פרוצדורה

  • פונקציה: מחזירה ערך יחיד (חובה RETURN), נקראת בתוך SELECT/WHERE, כל הפרמטרים קלט בלבד.
  • פרוצדורה: מופעלת ב-CALL, יכולה לבצע כמה פעולות, להחזיר דרך OUT או להציג SELECT, ורשאית לשנות נתונים (INSERT/UPDATE).

מגבלות הפונקציה מהמצגת: חייבת להחזיר ערך; פרמטרים הם קלט-בלבד; בדרך-כלל אינה משנה טבלאות ואינה שולטת בטרנזקציות (commit/rollback) — לכל אלה משתמשים בפרוצדורה.

🧪 מעבדת SQL — תרגול חי

כתבו SQL אמיתי והריצו אותו כאן, ישירות בדפדפן. ▶ הרצה מציגה את התוצאה שלכם; בדיקה מול פתרון משווה אותה לתוצאת הפתרון הרשמי.

המנוע (SQLite ב-WebAssembly) רץ כולו בדפדפן ותואם ל-MySQL בכל מה שנלמד בקורס — AUTO_INCREMENT, DESCRIBE ו-TRUNCATE מתורגמים אוטומטית. כל הרצה מתחילה ממסד נקי, אז אי-אפשר לקלקל כלום.

🧪 משימה: צרו VIEW בשם high_earners עם שם ושכר של העובדים שמרוויחים מעל 8000 — ואז שלפו ממנו את כל השורות.

CREATE VIEW high_earners AS
SELECT first_name, salary FROM employee
WHERE salary > 8000;

SELECT * FROM high_earners;

🧪 משימה: סנדבוקס חופשי — נסו כאן כל דבר על טבלת employee. הערה: פונקציות ופרוצדורות מאוחסנות אינן נתמכות במנוע הדפדפן (SQLite) — אותן מתרגלים ב-MySQL אמיתי; Views נתמכים במלואם.
