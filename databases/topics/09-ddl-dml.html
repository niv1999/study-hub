DDL — הגדרת מבנה

DDL (Data Definition Language) מגדיר את מבנה המסד:

CREATE TABLE employee (
  emp_id     INT PRIMARY KEY AUTO_INCREMENT,
  first_name VARCHAR(50) NOT NULL,
  salary     DECIMAL(10,2),
  dept_id    INT,
  FOREIGN KEY (dept_id) REFERENCES department(dept_id)
);

ALTER TABLE employee ADD email VARCHAR(100);   -- הוספת עמודה
ALTER TABLE employee DROP email;               -- הורדת עמודה
ALTER TABLE employee MODIFY first_name VARCHAR(80);  -- שינוי טיפוס
DROP TABLE employee;   -- מוחק את הטבלה כולה

AUTO_INCREMENT נותן למפתח ערך רץ אוטומטי (1,2,3…) בכל הוספה. לצפייה במבנה טבלה: DESCRIBE tablename; (מציג שם עמודה, טיפוס, Null, Key, Default). אפשר גם ליצור טבלה משאילתה: CREATE TABLE t AS SELECT …;

מקור: ddl.sql

טיפוסי נתונים נפוצים

  • INT / BIGINT — מספרים שלמים.
  • DECIMAL(p,s) — מספר עשרוני מדויק (כסף). FLOAT/DOUBLE — קירוב.
  • VARCHAR(n) — טקסט משתנה עד n. CHAR(n) — קבוע. TEXT — ארוך.
  • DATE, DATETIME, TIMESTAMP, BOOLEAN.

כולם ביחד בטבלה אחת:

CREATE TABLE product (
  product_id INT PRIMARY KEY AUTO_INCREMENT,
  name       VARCHAR(100) NOT NULL,     -- טקסט באורך משתנה, חובה
  sku        CHAR(8),                   -- קוד באורך קבוע
  price      DECIMAL(8,2),              -- כסף: מדויק, 2 ספרות אחרי הנקודה
  weight     FLOAT,                     -- קירוב — לא לכסף!
  in_stock   BOOLEAN DEFAULT TRUE,
  added_at   DATETIME
);

DML — שינוי נתונים

-- תבניות:
-- INSERT INTO table (cols) VALUES (vals);
-- UPDATE table SET col = val WHERE condition;
-- DELETE FROM table WHERE condition;

INSERT INTO employee (emp_id, first_name, salary)
VALUES (1, 'Dana', 8000);

INSERT INTO employee (first_name, salary)      -- כמה שורות בבת אחת;
VALUES ('Noa', 9000), ('Avi', 6500);           -- emp_id יושלם מ-AUTO_INCREMENT

UPDATE employee SET salary = salary * 1.1      -- העלאה של 10%
WHERE dept_id = 10;                            -- רק למחלקה 10

DELETE FROM employee WHERE emp_id = 1;         -- מוחק את Dana בלבד

מסוכן: UPDATE/DELETE ללא WHERE משנים/מוחקים את כל השורות. תמיד לוודא WHERE.

מקור: ddl_insert_update_delete.pdf

DROP מול DELETE מול TRUNCATE

  • DELETE FROM t; — DML, מוחק שורות (עם WHERE אופציונלי), ניתן לביטול בטרנזקציה.
  • TRUNCATE TABLE t; — DDL, מרוקן את כל הטבלה מהר, מאפס AUTO_INCREMENT, לרוב לא ניתן לביטול.
  • DROP TABLE t; — DDL, מוחק את הטבלה עצמה (מבנה + נתונים).
DELETE FROM employee WHERE dept_id = 20;  -- מוחק רק את עובדי מחלקה 20
DELETE FROM employee;                     -- הטבלה קיימת אך ריקה

TRUNCATE TABLE employee;                  -- ריקון מהיר + איפוס המונה של AUTO_INCREMENT

DROP TABLE employee;                      -- הטבלה איננה —
SELECT * FROM employee;                   -- ERROR: Table ... doesn't exist

🧪 מעבדת SQL — תרגול חי

כתבו SQL אמיתי והריצו אותו כאן, ישירות בדפדפן. ▶ הרצה מציגה את התוצאה שלכם; בדיקה מול פתרון משווה אותה לתוצאת הפתרון הרשמי.

המנוע (SQLite ב-WebAssembly) רץ כולו בדפדפן ותואם ל-MySQL בכל מה שנלמד בקורס — AUTO_INCREMENT, DESCRIBE ו-TRUNCATE מתורגמים אוטומטית. כל הרצה מתחילה ממסד נקי, אז אי-אפשר לקלקל כלום.

🧪 משימה: צרו טבלת product עם: product_id מפתח ראשי, name טקסט חובה, price עשרוני — והכניסו שני מוצרים.

CREATE TABLE product (
  product_id INT PRIMARY KEY,
  name  VARCHAR(100) NOT NULL,
  price DECIMAL(8,2)
);
INSERT INTO product VALUES (1,'Laptop',4500.00),(2,'Mouse',89.90);

🧪 משימה: העלו ב-10% את השכר של עובדי מחלקה 10 בלבד (UPDATE עם WHERE).

UPDATE employee SET salary = salary * 1.1
WHERE dept_id = 10;

DELETE FROM employee WHERE dept_id IS NULL;
