אילוצים (Constraints)

  • PRIMARY KEY — ייחודי + לא-NULL, מזהה שורה.
  • FOREIGN KEY ... REFERENCES — שלמות התייחסות לטבלה אחרת.
  • UNIQUE — ערכים ייחודיים (אך NULL מותר, ולרוב מרובה).
  • NOT NULL — חובה ערך. DEFAULT val — ערך ברירת מחדל.
  • CHECK (condition) — כל שורה חייבת לקיים את התנאי.

כולם ביחד — הדוגמה מהשיעור (employee עם שכר מינימום):

CREATE TABLE employee (
  emp_id   INT PRIMARY KEY,                 -- ייחודי + NOT NULL
  emp_ssn  CHAR(9) NOT NULL UNIQUE,         -- ת"ז: חובה וייחודי (מפתח מועמד)
  emp_name VARCHAR(40) NOT NULL,            -- חובה, אבל לא ייחודי
  status   VARCHAR(10) DEFAULT 'active',    -- ברירת מחדל אם לא סופק ערך
  pay_rate DECIMAL(5,2) NOT NULL,
  CHECK (pay_rate > 5.25),
  CHECK (status IN ('active', 'vacation', 'suspended'))
);

-- INSERT שמפר אילוץ — נדחה:
INSERT INTO employee (emp_id, emp_ssn, emp_name, pay_rate)
VALUES (1, '123456789', 'Dana', 3.00);
-- ERROR: Check constraint violated   (3.00 < 5.25)

INSERT INTO employee (emp_id, emp_ssn, emp_name, pay_rate)
VALUES (2, '123456789', 'Avi', 40.00);   -- אחרי שהוכנס עובד עם אותה ת"ז
-- ERROR: Duplicate entry for key emp_ssn (UNIQUE)

CHECK ו-NULL

CREATE TABLE employee (
  emp_id INT PRIMARY KEY,
  salary DECIMAL(10,2) CHECK (salary > 0),
  age    INT CHECK (age >= 18)
);

CHECK «עובר» כאשר התנאי אינו FALSE — כלומר גם על NULL (שמעריך ל-UNKNOWN). כדי לחייב ערך, מוסיפים גם NOT NULL.

ON DELETE / ON UPDATE

התנהגות מפתח זר כשהאב משתנה/נמחק:

  • ON DELETE CASCADE — מחיקת האב מוחקת את הבנים.
  • ON DELETE SET NULL — המפתח הזר בבנים הופך ל-NULL.
  • ON DELETE RESTRICT/NO ACTION — מונע מחיקת אב שיש לו בנים (ברירת מחדל).

הדוגמה מהשיעור — תוכנית בריאות שמפנה לעובד:

CREATE TABLE health_plan (
  emp_ssn  CHAR(9) PRIMARY KEY,
  provider VARCHAR(20) NOT NULL,
  FOREIGN KEY (emp_ssn) REFERENCES employee(emp_ssn)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

DELETE FROM employee WHERE emp_id = 1;
-- עם CASCADE:  שורת ה-health_plan של אותו עובד נמחקת גם היא
-- בלי (ברירת המחדל RESTRICT): המחיקה נכשלת —
-- ERROR: Cannot delete a parent row: a foreign key constraint fails

טריגרים (Triggers)

טריגר הוא קוד שרץ אוטומטית בתגובה ל-INSERT/UPDATE/DELETE, לפני (BEFORE) או אחרי (AFTER) הפעולה. גישה לערכים דרך NEW (השורה החדשה) ו-OLD (הישנה):

CREATE TRIGGER trg_upper
BEFORE INSERT ON employee
FOR EACH ROW
SET NEW.first_name = UPPER(NEW.first_name);

BEFORE מאפשר לשנות/לאמת את NEW לפני השמירה; AFTER שימושי ללוגים/עדכון טבלאות אחרות.

🧪 מעבדת SQL — תרגול חי

כתבו SQL אמיתי והריצו אותו כאן, ישירות בדפדפן. ▶ הרצה מציגה את התוצאה שלכם; בדיקה מול פתרון משווה אותה לתוצאת הפתרון הרשמי.

המנוע (SQLite ב-WebAssembly) רץ כולו בדפדפן ותואם ל-MySQL בכל מה שנלמד בקורס — AUTO_INCREMENT, DESCRIBE ו-TRUNCATE מתורגמים אוטומטית. כל הרצה מתחילה ממסד נקי, אז אי-אפשר לקלקל כלום.

🧪 משימה: צרו טבלת account עם account_id מפתח ראשי ו-balance עשרוני עם CHECK שאוסר יתרה שלילית, והכניסו חשבון אחד חוקי. אחר-כך נסו להריץ גם INSERT עם יתרה שלילית — תקבלו שגיאה (לפני «בדיקה» השאירו רק את החוקי).

CREATE TABLE account (
  account_id INT PRIMARY KEY,
  balance DECIMAL(10,2) CHECK (balance >= 0)
);
INSERT INTO account VALUES (1, 100.00);

🧪 משימה: צרו טריגר שאחרי INSERT ל-employee מוסיף את שם העובד החדש (NEW.first_name) לטבלת log — ואז הכניסו את העובדת (1, 'Dana').

CREATE TRIGGER trg_new_emp
AFTER INSERT ON employee
FOR EACH ROW
BEGIN
  INSERT INTO log (msg) VALUES (NEW.first_name);
END;

INSERT INTO employee VALUES (1, 'Dana');

🧪 משימה: בטבלאות יש עובדים ותוכניות בריאות עם ON DELETE CASCADE. מחקו את העובד עם emp_id = 1 — וראו איך שורת ה-health_plan שלו נעלמת מעצמה.

DELETE FROM employee WHERE emp_id = 1;
