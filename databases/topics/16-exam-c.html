employee(emp_id, first_name, last_name, salary, hire_date,

חלק 2 — שאלות פתוחות (כתיבת SQL)

נסו לכתוב את השאילתה בעצמכם, ואז גלו את הפתרון לדוגמה.

שאלה פתוחה 1

כתבו שאילתה המחזירה לכל מיקום (location) את מספר העובדים המועסקים במחלקות שבאותו מיקום, רק למיקומים עם מעל 20 עובדים.

SELECT d.location, COUNT(e.emp_id) AS num_employees
FROM department d
JOIN employee e ON e.dept_id = d.dept_id
GROUP BY d.location
HAVING COUNT(e.emp_id) > 20
ORDER BY num_employees DESC;

מקבצים לפי location (ולא לפי מחלקה), סופרים עובדים, ומסננים את הקבוצות עם HAVING.

🧪 משימה: כתבו את הפתרון שלכם כאן והריצו אותו על נתוני דוגמה אמיתיים — ורק אז השוו לפתרון. (בנתונים: בתל אביב 28 עובדים, בחיפה 4 — רק תל אביב אמורה לעבור את הסינון.)

SELECT d.location, COUNT(e.emp_id) AS num_employees
FROM department d
JOIN employee e ON e.dept_id = d.dept_id
GROUP BY d.location
HAVING COUNT(e.emp_id) > 20
ORDER BY num_employees DESC;

שאלה פתוחה 2

טבלת works_on(emp_id, proj_id, hours) — הסבירו מהו המפתח הראשי המתאים ולמה, וכתבו את פקודת ה-DDL ליצירתה עם המפתחות הזרים.

מפתח ראשי: הצירוף (emp_id, proj_id) — כי עובד יכול לעבוד על כמה פרויקטים ופרויקט מעסיק כמה עובדים, אך זוג עובד-פרויקט מופיע פעם אחת. זו טבלת קשר ל-N:M.

CREATE TABLE works_on (
  emp_id  INT,
  proj_id INT,
  hours   DECIMAL(5,1),
  PRIMARY KEY (emp_id, proj_id),
  FOREIGN KEY (emp_id)  REFERENCES employee(emp_id),
  FOREIGN KEY (proj_id) REFERENCES project(proj_id)
);

🧪 משימה: כתבו את ה-CREATE TABLE של works_on (מפתח ראשי מורכב + שני מפתחות זרים) והריצו; «בדיקה» מוודאת שיש 2 עמודות מפתח ו-2 מפתחות זרים.

CREATE TABLE works_on (
  emp_id  INT,
  proj_id INT,
  hours   DECIMAL(5,1),
  PRIMARY KEY (emp_id, proj_id),
  FOREIGN KEY (emp_id)  REFERENCES employee(emp_id),
  FOREIGN KEY (proj_id) REFERENCES project(proj_id)
);
