חלק 2 — שאלות פתוחות (כתיבת SQL)

נסו לכתוב את השאילתה בעצמכם, ואז גלו את הפתרון לדוגמה.

שאלה פתוחה 1

כתבו שאילתה המחזירה את שם המחלקה ואת שם העובד המשתכר ביותר בה (לכל מחלקה שורה אחת). התמודדו עם המקרה של שוויון בשכר.

SELECT d.dept_name, e.first_name, e.last_name, e.salary
FROM employee e
JOIN department d ON e.dept_id = d.dept_id
WHERE e.salary = (
  SELECT MAX(e2.salary) FROM employee e2
  WHERE e2.dept_id = e.dept_id
);

תת-שאילתה מתואמת מוצאת את השכר המקסימלי בכל מחלקה. אם יש שוויון — כל העובדים ששכרם שווה למקסימום יוחזרו (זה תקין; אם רוצים אחד בלבד אפשר LIMIT עם מיון או פונקציית חלון).

🧪 משימה: כתבו את הפתרון שלכם כאן והריצו אותו על נתוני דוגמה אמיתיים — ורק אז השוו לפתרון. (בנתונים: המרוויחה הגדולה ב-R&D היא Dana וב-HR — Noa.)

SELECT d.dept_name, e.first_name, e.last_name, e.salary
FROM employee e
JOIN department d ON e.dept_id = d.dept_id
WHERE e.salary = (
  SELECT MAX(e2.salary) FROM employee e2
  WHERE e2.dept_id = e.dept_id
);

שאלה פתוחה 2

נתונה טבלה לא-מנורמלת orders(order_id, customer_name, customer_phone, product_name, product_price, qty). פרקו ל-3NF (ציינו טבלאות, מפתחות ראשיים וזרים) והסבירו איזו אנומליה נמנעת.

Customer(customer_id PK, customer_name, customer_phone)
Product(product_id PK, product_name, product_price)
Orders(order_id PK, customer_id FK -> Customer)
OrderLine(order_id FK -> Orders,
          product_id FK -> Product,
          qty,
          PRIMARY KEY (order_id, product_id))

אנומליות שנמנעות: ללא הפירוק, מחיר מוצר חוזר בכל שורת הזמנה (כפילות); שינוי מחיר ידרוש עדכון שורות רבות (update anomaly); מחיקת ההזמנה האחרונה של מוצר תמחק את פרטי המוצר (delete anomaly); ואי אפשר להוסיף מוצר חדש בלי הזמנה (insert anomaly). הפירוק שומר כל עובדה במקום אחד.

🧪 משימה: ממשו את הפירוק שלכם ב-DDL: צרו את הטבלאות המנורמלות עם מפתחות ראשיים וזרים. «בדיקה» מוודאת שנוצרו 4 טבלאות (כמו בפתרון: customer, product, orders, order_line).

CREATE TABLE customer (
  customer_id INT PRIMARY KEY,
  customer_name VARCHAR(50),
  customer_phone VARCHAR(20)
);
CREATE TABLE product (
  product_id INT PRIMARY KEY,
  product_name VARCHAR(50),
  product_price DECIMAL(8,2)
);
CREATE TABLE orders (
  order_id INT PRIMARY KEY,
  customer_id INT,
  FOREIGN KEY (customer_id) REFERENCES customer(customer_id)
);
CREATE TABLE order_line (
  order_id INT, product_id INT, qty INT,
  PRIMARY KEY (order_id, product_id),
  FOREIGN KEY (order_id) REFERENCES orders(order_id),
  FOREIGN KEY (product_id) REFERENCES product(product_id)
);
